Introduction

Sortable DL is a jQuery plugin to turn a definition list into a sortable table. Sortable DL was born out of the need for a sortable table with the ability to open a toggle layer under a corresponding table row. Choose this plugin if you need something similar. If you just need a sortable table functionality, check out the awesome solutions out there.

Documentation

To use Sortabel DL include jQuery and the Sortable DL plugin inside the <head> area of your HTML document.

<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.4/jquery.min.js"></script>
<script src="sortable-dl.min.js"></script>

Insert a configuration script in the <head> area of your HTML document.

<script>
var show = function(num) {
  $('#sortable dl').each(function() {
	var dd = $(this).children('dd');
	var tr = $(this).parents('tr');
	if ( (tr.attr('id') === 'row' + num) && $(this).children('dd').css('display') === 'none' ) {
      $(this).children('dd').removeClass('hide').addClass('ib');
    } else {
      $(this).children('dd').removeClass('ib').addClass('hide');
    }
  });
};

$(document).ready(function() {  

  var mySorter = new SortableDL();
  mySorter.init({
    id: 'sortable',       // Table ID
    first: 'ascending',   // optional / default 'descending'
    columns: [
      null,               // Col 1
      'caseInsensitive',  // Col 2
      'caseSensitive',    // Col 3
      'caseInsensitive',  // Col 4
      'numeric',          // Col 5
      'date',             // Col 6  
      null                // Col 7
    ],
    classes: {
      asc: 'asc',         // optional / default: 'asc'
      desc: 'desc',       // optional / default: 'desc'
      column: 'active-column' // optional / default: ''
    },
    pagination: {
	  items: 5,           // Items per page
	  numbers: 5          // Visible page numbers
    }
  });  

});
</script>
